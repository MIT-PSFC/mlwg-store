# Things in <> can be replaced with any name of your choice

# If you wish to use GPUs, follow additional steps labelled with ~
#   These are only intended to work with NVIDIA with CUDA 12.4.1, so make sure that is installed

# Install pixi (Linux)
cd
curl -fsSL https://pixi.sh/install.sh | bash

# Restart your terminal for pixi commands to be recognized

# Create and setup the virtual env using pixi
cd <work_area>
pixi init <env_name>
cd <env_name>

# ~ Add package channels to generated environment file, pixi.toml
channels = ["nvidia/label/12.4.1", "conda-forge", "pytorch"]

# Install recommended libraries
pixi add git
pixi add 'python>=3.12,<3.13'
pixi add pip
pixi add numpy
pixi add scipy
pixi add pandas
pixi add xarray
pixi add hdf5
pixi add h5py
pixi add netcdf4
pixi add pytables
pixi add scikit-learn
pixi add ipython
pixi add greenlet
pixi add nbformat
pixi add ipykernel

# If using CPUs
pixi add pytorch
pixi add matplotlib
pixi add plotly

# ~ If using GPUs
pixi add 'cuda-toolkit>=12.4.1,<12.4.2'
pixi add pytorch-gpu
pixi add cudnn
pixi add --pypi matplotlib
pixi add --pypi plotly

# To enter environment, make sure you are in some subdirectory of <env_name> and use:
pixi shell

# To exit environment, simply use:
exit

# When in environment, you can run any of our tutorials with:
python <script_name>
